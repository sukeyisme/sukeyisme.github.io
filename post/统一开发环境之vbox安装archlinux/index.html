统一开发环境之virtualbox安装archlinux

 Posted on January 8, 2018

缘由

每次换开发机，最痛苦的就是重新配置开发环境，最要命的是每次配环境都没有记录。这次狠下心，决定记录下过程。为以后配置环境做参考。为开发配置环境已经折腾了无数次了 ，这次一定是最后一次。

为什么选择virtualbox

我是在mac上开发的，之前用过paralles、vmware fusion安装过arch，原生都不支持arch。再加上arch是滚动更新的，有时候Guest Additions 跟不上arch的节奏，导致某些功能失效或报错。 很是折腾，决定弃用，不想再折腾了。

为什么选择archlinux

硬盘有限，只想少装点默认的东西。滚动更新，方便的包管理工具pacman 等等。

虚拟机创建过程

虚拟硬盘文件类型选择VMDK

安装arch

    #默认us键盘不需要设置

    #使用默认源 也不需要修改
    # gdisk /dev/sda
      Command (? for help): n
      Partition number (1-128, default 1): ENTER
      First Sector [...]: ENTER
      Last Sector [...]: ENTER
      Current type is 'Linux filesystem'
      Hex code or GUID (L to show codes, Enter = 8300): ENTER
      Changed type of partition to 'Linux filesystem'
    - w (write)
    - y (confirm write)

    # mkfs.ext4 /dev/sda1
    # mount /dev/sda1 /mnt
    # pacstrap /mnt base
    # genfstab -p /mnt > /mnt/etc/fstab
    # arch-chroot /mnt

    # echo archlinux > /etc/hostname
    # echo LANG=en_US.UTF-8 > /etc/locale.conf
    # echo LC_COLLATE=C >> /etc/locale.conf

    # rm /etc/localtime
    # ln -s /usr/share/zoneinfo/Asia/Shanghai /etc/localtime

    # nano /etc/locale.gen
    - uncomment:
      en_US.UTF-8 UTF-8
      zh_CN.UTF-8 UTF-8
      和其他zh_CN 的选项
    - ctrl+x (exit)
    - y (confirm write)
    # locale-gen

    # nano /etc/pacman.conf
    - uncomment:
      [multilib]
      Include = /etc/pacman.d/mirrorlist
    - ctrl+x (exit)
    - y (confirm write)

    # 因为需要需要图形界面所以安装virtualbox-guest-utils 如果不需要的话安装virtualbox-guest
    # pacman -Sy gptfdisk syslinux xz sudo networkmanager openssh ntp virtualbox-guest-utils
    - y (confirm)

    #使用lts版本内核
    #pacman -R linux

    # nano /etc/mkinitcpio.conf
    - change:
      HOOKS="base udev autodetect ..."
    - to:
      HOOKS="systemd autodetect ..."
    - uncomment:
      COMPRESSION="xz"
    - uncomment and change:
      COMPRESSION_OPTIONS=()
    - to:
      COMPRESSION_OPTIONS="-9"
    - ctrl+x (exit)
    - y (confirm write)

    # mkinitcpio -p linux-lts
    # passwd
      Enter new UNIX password: vagrant
      Retype new UNIX password: vagrant
      passwd: password updated successfully

    # 吧syslinux.cfg 中的linux 都 加上-lts 因为内核改成lts版本了
    # nano /boot/syslinux/syslinux.cfg
    - change:
      APPEND root=/dev/sda3 rw
    - to:
      APPEND root=/dev/sda1 rw
    - comment following labels:
      archfallback, hdt, reboot and poweroff
    - ctrl+x (exit)
    - y (confirm write)

    # syslinux-install_update -iam

    # nano /etc/sudoers
    - uncomment:
      %wheel ALL=(ALL) NOPASSWD: ALL
    - ctrl+x (exit)
    - y (confirm write)

    # nano /etc/ntp.conf
    - change:
      server [0-3].arch.pool.ntp.org
    - to:
      server [0-3].cn.pool.ntp.org
    # ntpd -gq

    # useradd -m -g users -s /bin/bash vagrant
    # gpasswd -a vagrant wheel
    # gpasswd -a vagrant vboxsf

    # passwd vagrant
      Enter new UNIX password: vagrant
      Retype new UNIX password: vagrant
      passwd: password updated successfully

    # systemctl enable sshd
    # systemctl enable vboxservice
    # systemctl enable rpcbind
    # systemctl enable NetworkManager

    # mkdir -p /home/vagrant/.ssh
    # pacman -S wget
    # wget -L -O /home/vagrant/.ssh/authorized_keys https://raw.githubusercontent.com/hashicorp/vagrant/master/keys/vagrant.pub
    # chmod 0600 /home/vagrant/.ssh/authorized_keys
    # chmod 0700 /home/vagrant/.ssh
    # chown -R vagrant:users /home/vagrant

    # exit
    # umount -R /mnt
    # systemctl reboot

    - cleanup, login as vagrant
    # sudo su
    # pacman -Scc --noconfirm
    # rm -rf /var/log/journal/* /var/log/old/* /var/log/faillog /var/log/lastlog /var/log/pacman.log
    # rm -f /home/vagrant/.bash_history
    # rm -f /root/.bash_history
    # cd /root
    # dd if=/dev/zero of=zerofillfile bs=1M
    # rm -f zerofillfile
    # history -c

    # systemctl poweroff
